def connections(filter={})
  begin 
    response = RestClient.get "#{@baseurl}/#{filter[:id]}/#{filter[:connection]}/", 
    { params: { access_token: @access_token } }
    result = JSON.parse(response)
    return result['data']
  rescue => e
    e.response
  end
end
